ConfigurarRede() {

echo "ifconfig"

echo "Insira sua interface de rede"

read interfacerede

echo ifconfig $interfacerede down

echo "Insira seu endereço correto da rede (inet end)"

read endereco

echo "Insira o netmask (Masc)"

read netmaske

echo "ifconfig $interfacerede $endereco netmask $netmaske up"

echo "SUCESSO !"

echo "Deseja configurar o seu DNS agora ? (s/n) "

read sn

if [ "$sn"=="s" ]

then

AlterarDNSConfigurarRede

else

echo "Lembre-se de configurar seu DNS"

fi

exit

}

AlterarDNSConfigurarRede() {

gedit /etc/resolv.conf

echo "Editado com sucesso."

DNS

}

GerenciarUsuarios() {

echo

echo -n "Qual a opção desejada? "

read opcao

case $opcao in

1) Adicionar ;;

2) ModSenhaUser ;;

3) Deletar ;;

4) AdicionarUserGroup ;;

5) Removerusergroup ;;

6) CriaGrupo ;;

7) RemGrupo ;;

8) VoltarPrincipal ;;

*) "Opção desconhecida." ; echo ; Principal ;;

esac

}

Adicionar() {

clear

echo -n "Digite o nome do novo usuário:"

read nome

adduser $nome

Principal

}

ModSenhaUser() {

clear

echo -n "Digite o nome do usuário para alterar a senha:"

read senha

passwd $senha

Principal

}

Deletar() {

clear

echo -n "Digite o nome do usuário a deletar:"

read nome

userdel $nome

echo "------------------------------------------"

echo "Usuário $nome removido com sucesso !!!"

echo "------------------------------------------"

Principal

}

AdicionarUserGroup() {

echo "Digite o usuário a ser adicionado em um grupo"

read usuarioparagrupo

echo "Digite o grupo que o $usuarioparagrupo irá entrar"

read groupentrar

usermod -G $groupentrar $usuarioparagrupo

clear

echo "$usuarioparagrupo Adicionado no grupo $groupentrar com sucesso !!! "

GerenciarUsuarios

}

Removerusergroup() {

echo "Digite o usuário a ser removido de um grupo"

read usuariosair

echo "Digite o grupo que o $usuariosair irá sair"

read groupasair

gpasswd -d $usuariosair $groupasair

clear

echo "$usuariosair removido do grupo $groupasair com sucesso !!! "

GerenciarUsuarios

}

RemGrupo() {

compactarTar()

{

clear

echo -n "Digite o diretorio onde esta o arquivo:"

read DIRETORIO

echo -n "Digite o nome do arquivo: "

read ARQUIVO

COMPACTAR="$DIRETORIO/$ARQUIVO"

tar -cf $ARQUIVO.tar $COMPACTAR 2> /dev/null

if [ $? -eq 0 ]; then

echo "Arquivo compactado"

sleep 2

clear

else

echo "Arquivo não foi compactado"

clear

fi

Principal

}
